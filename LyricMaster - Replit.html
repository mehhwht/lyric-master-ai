    import React, { useState } from "react";
    import { Button } from "@/components/ui/button";

    export default function MusicStudio() {
    const [blocks, setBlocks] = useState([
    { id: 1, type: "vocals", label: "🎤 Vocals" },
    { id: 2, type: "drums", label: "🥁 Drums" },
    { id: 3, type: "melody", label: "🎹 Melody" },
    ]);

    const handleDragStart = (e, id) => {
    e.dataTransfer.setData("blockId", id);
    };

    const handleDrop = (e) => {
    const blockId = e.dataTransfer.getData("blockId");
    const draggedBlock = blocks.find((b) => b.id === parseInt(blockId));
    const updated = [...blocks.filter((b) => b.id !== draggedBlock.id), draggedBlock];
    setBlocks(updated);
    };

    return (
    <div className="p-6">
    <h1 className="text-3xl font-bold mb-4">🎶 AI MUSIC STUDIO</h1>
    <input className="border border-gray-400 rounded p-2 w-full mb-4" placeholder="Type your song idea here..." />
    <Button className="mb-6">Generate Song</Button>

      <div className="bg-gray-100 p-4 rounded shadow-md">
        <h2 className="text-xl font-semibold mb-3">🎛️ Drag and Drop Timeline</h2>
        <div
          className="grid grid-cols-3 gap-4"
          onDragOver={(e) => e.preventDefault()}
          onDrop={handleDrop}
        >
          {blocks.map((block) => (
            <div
              key={block.id}
              draggable
              onDragStart={(e) => handleDragStart(e, block.id)}
              className="bg-white p-3 rounded shadow hover:bg-blue-50 cursor-move"
            >
              {block.label}
            </div>
          ))}
        </div>
      </div>
    </div>

    );
    }

    from fastapi import FastAPI, File, UploadFile, Form
    from fastapi.responses import JSONResponse
    from fastapi.middleware.cors import CORSMiddleware
    import aiofiles
    import os
    import requests
    import re

    app = FastAPI()

    app.add_middleware(
    CORSMiddleware,
    allow_origins=[""],
    allow_credentials=True,
    allow_methods=["    "],
    allow_headers=["*"],
    )

    UPLOAD_DIR = "uploaded_audios"
    os.makedirs(UPLOAD_DIR, exist_ok=True)

    def get_drive_download_link(url: str) -> str:
    match = re.search(r"/d/([\w-]+)", url)
    if not match:
    return None
    file_id = match.group(1)
    return f"https://drive.google.com/uc?export=download&id={file_id}"

    @app.post("/upload/")
    async def upload_audio(file: UploadFile = File(None), drive_link: str = Form(None)):
    if file:
    filepath = os.path.join(UPLOAD_DIR, file.filename)
    async with aiofiles.open(filepath, 'wb') as out_file:
    content = await file.read()
    await out_file.write(content)
    return {"status": "success", "source": "upload", "filename": file.filename}

    elif drive_link:
        direct_url = get_drive_download_link(drive_link)
        if not direct_url:
            return JSONResponse(status_code=400, content={"error": "Invalid Google Drive link"})
        response = requests.get(direct_url, stream=True)
        if response.status_code != 200:
            return JSONResponse(status_code=400, content={"error": "Failed to fetch file from Drive. Make sure it is public."})
        filename = drive_link.split('/')[-1].split('?')[0] + ".mp3"
        filepath = os.path.join(UPLOAD_DIR, filename)
        with open(filepath, 'wb') as f:
            f.write(response.content)
        return {"status": "success", "source": "drive", "filename": filename}
    else:
        return JSONResponse(status_code=400, content={"error": "No file or link provided."})
